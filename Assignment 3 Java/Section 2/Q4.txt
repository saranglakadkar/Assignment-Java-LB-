public class DoWhileLoop { 
 public static void main(String[] args) { 
 int i = 1; 
 do { 
 System.out.print(i + " "); 
 i++; 
 } while (i < 5); 
 System.out.println(i); 
 } 
} 


// Guess the output of this do-while loop.

Output-> 1 2 3 4 (for do-while loop) after loop print 5 also.

1 2 3 4 5
